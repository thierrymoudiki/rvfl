Ridge Gamma model

library(rvfl)
glmGamma <- function(formula, ...) {
  stats::glm(formula = formula, 
             family = "gaussian",
             #family = inverse.gaussian(link = "1/mu^2"),
             #family = Gamma(link = "log"),
             #control = glm.control(maxit = 100, epsilon = 1e-8),
             ...)
}
